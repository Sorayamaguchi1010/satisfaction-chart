  <canvas id="satisfactionChart" width="400" height="400"></canvas>


  <script>

    const ctx = document.getElementById('satisfactionChart').getContext('2d');

    let chart = new Chart(ctx, {

      type: 'scatter',

      data: {

        datasets: [{

          label: '満足度',

          data: [{x: 25, y: 25}],

          backgroundColor: 'blue',

          pointRadius: 10,

        }]

      },

      options: {

        scales: {

          x: {

            min: 0,

            max: 50,

            title: { display: true, text: '自分の満足度合計 (0-50)' }

          },

          y: {

            min: 0,

            max: 50,

            title: { display: true, text: '相手の満足度合計 (0-50)' }

          }

        }

      }

    });


    function updateChart() {

      const myScores = [

        +document.getElementById('my1').value,

        +document.getElementById('my2').value,

        +document.getElementById('my3').value,

        +document.getElementById('my4').value,

        +document.getElementById('my5').value,

      ];

      const partnerScores = [

        +document.getElementById('partner1').value,

        +document.getElementById('partner2').value,

        +document.getElementById('partner3').value,

        +document.getElementById('partner4').value,

        +document.getElementById('partner5').value,

      ];

      const myTotal = myScores.reduce((a,b)=>a+b,0);

      const partnerTotal = partnerScores.reduce((a,b)=>a+b,0);


      chart.data.datasets[0].data = [{x: myTotal, y: partnerTotal}];

      chart.update();

    }

  </script>
